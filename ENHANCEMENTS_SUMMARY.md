## 📚 Usage Examples

### Adding Animations to Pages
```tsx
import { ParallaxSection } from '@/components/animations/parallax-section'
import { GlassCard } from '@/components/animations/glass-card'

<ParallaxSection speed={0.5}>
  <GlassCard hover={true}>
    {/* Your content */}
  </GlassCard>
</ParallaxSection>
```

### Using Toast Notifications
```tsx
import { toast } from '@/components/animations/toast-notification'

toast('Achievement unlocked!', 'success', 3000)
```

### Animated Statistics
```tsx
import { AnimatedStat } from '@/components/animations/animated-stats'

<AnimatedStat 
  value={5000} 
  label="Active Students" 
  suffix="+" 
  duration={2}
/>
```
